const HABITS = [
  {id:'libro1',name:'Leer 20 págs — Libro 1',cat:'Lectura',icon:'📖',xp:15,priority:2},
  {id:'libro2',name:'Leer 20 págs — Libro 2',cat:'Lectura',icon:'📚',xp:15,priority:2},
  {id:'dieta',name:'Cumplir con la dieta',cat:'Alimentación',icon:'🥗',xp:20,priority:3},
  {id:'agua',name:'Beber 2L de agua',cat:'Salud',icon:'💧',xp:10,priority:2},
  {id:'dormir',name:'Dormir 7-8 horas',cat:'Salud',icon:'💤',xp:15,priority:3},
  {id:'ejercicio',name:'Ejercicio 60 min',cat:'Salud',icon:'🏋️',xp:25,priority:3},
  {id:'meditacion',name:'Meditación 10 min',cat:'Mente',icon:'🧘',xp:10,priority:2},
  {id:'redes',name:'Sin redes por la mañana',cat:'Mente',icon:'📵',xp:10,priority:2},
  {id:'gratitud',name:'Agradecer 3 cosas',cat:'Mente',icon:'🙏',xp:10,priority:1},
];

const CAT_CLASS={Lectura:'cat-lectura',Salud:'cat-salud',Mente:'cat-mente',Alimentación:'cat-alim'};
const CAT_COLOR={Lectura:'#3b82f6',Salud:'#10b981',Mente:'#a78bfa',Alimentación:'#f59e0b'};
const LEVELS=[{n:'Aprendiz',xp:0},{n:'Constante',xp:200},{n:'Disciplinado',xp:500},{n:'Imparable',xp:1000},{n:'Maestro',xp:2000}];
const todayKey=new Date().toISOString().split('T')[0];

function loadState(){try{return JSON.parse(localStorage.getItem('ht_state')||'{}')}catch{return{}}}
function saveState(s){localStorage.setItem('ht_state',JSON.stringify(s))}

let state=loadState();
if(!state.history)state.history={};
if(!state.xp)state.xp=0;
if(!state.history[todayKey])state.history[todayKey]=[];

function todayDone(){return state.history[todayKey]||[]}
function isChecked(id){return todayDone().includes(id)}

function toggle(id){
  let d=todayDone();
  if(d.includes(id)){
    state.history[todayKey]=d.filter(x=>x!==id);
    const h=HABITS.find(h=>h.id===id);
    state.xp=Math.max(0,state.xp-(h?h.xp:10));
  }else{
    state.history[todayKey]=[...d,id];
    const h=HABITS.find(h=>h.id===id);
    state.xp=(state.xp||0)+(h?h.xp:10);
  }
  saveState(state);render();
}

function getStreak(id){
  let streak=0;
  for(let i=0;i<365;i++){
    const k=new Date(Date.now()-i*864e5).toISOString().split('T')[0];
    if(state.history[k]&&state.history[k].includes(id))streak++;
    else if(i>0)break;
  }
  return streak;
}

function getGlobalStreak(){
  let streak=0;
  for(let i=0;i<365;i++){
    const k=new Date(Date.now()-i*864e5).toISOString().split('T')[0];
    if((state.history[k]||[]).length===HABITS.length)streak++;
    else if(i>0)break;
  }
  return streak;
}

function getLevel(xp){let l=LEVELS[0];for(const lv of LEVELS)if(xp>=lv.xp)l=lv;return l;}
function getNextLevel(xp){for(const lv of LEVELS)if(xp<lv.xp)return lv;return null;}
function getWeekPct(){let t=0,d=0;for(let i=0;i<7;i++){const k=new Date(Date.now()-i*864e5).toISOString().split('T')[0];t+=HABITS.length;d+=(state.history[k]||[]).length;}return t?Math.round(d/t*100):0;}

function render(){
  const done=todayDone();
  const pct=Math.round(done.length/HABITS.length*100);
  document.getElementById('pctHoy').textContent=pct+'%';
  document.getElementById('doneCount').textContent=done.length+'/'+HABITS.length+' hábitos';
  document.getElementById('rachaGlobal').textContent='🔥'+getGlobalStreak();
  document.getElementById('xpTotal').textContent=state.xp;
  document.getElementById('semanaLabel').textContent=getWeekPct()+'%';

  const lv=getLevel(state.xp),nextLv=getNextLevel(state.xp);
  document.getElementById('levelLabel').textContent=lv.n;
  document.getElementById('xpBarLabel').textContent=state.xp+' XP';
  if(nextLv){
    document.getElementById('xpBarNext').textContent=nextLv.xp+' XP';
    document.getElementById('xpNext').textContent='Sig. nivel: '+nextLv.xp;
    document.getElementById('xpFill').style.width=Math.round((state.xp-lv.xp)/(nextLv.xp-lv.xp)*100)+'%';
  }else{
    document.getElementById('xpFill').style.width='100%';
    document.getElementById('xpBarNext').textContent='MAX';
    document.getElementById('xpNext').textContent='Nivel máximo';
  }

  const cats=[...new Set(HABITS.map(h=>h.cat))];
  const listEl=document.getElementById('habitList');
  listEl.innerHTML='';
  cats.forEach(cat=>{
    const div=document.createElement('div');
    div.innerHTML=`<div class="cat-header">${cat}</div>`;
    HABITS.filter(h=>h.cat===cat).forEach(h=>{
      const checked=done.includes(h.id),streak=getStreak(h.id);
      div.innerHTML+=`<div class="habit-item ${checked?'done':''}" id="hi-${h.id}">
        <button class="check-btn" onclick="toggle('${h.id}')">${checked?'✓':''}</button>
        <div class="habit-info">
          <div class="habit-name">${h.icon} ${h.name}</div>
          <div class="habit-meta" style="display:flex;gap:6px;align-items:center;margin-top:3px">
            <span class="cat-badge ${CAT_CLASS[h.cat]}">${h.cat}</span>
            ${streak>0?`<span class="habit-streak">🔥 ${streak}</span>`:''}
          </div>
        </div>
        <span class="habit-xp">+${h.xp} XP</span>
      </div>`;
    });
    listEl.appendChild(div);
  });

  const circ=2*Math.PI*55;
  document.getElementById('ringCircle').style.strokeDashoffset=circ*(1-pct/100);
  document.getElementById('ringPct').textContent=pct+'%';

  const catEl=document.getElementById('catStats');
  catEl.innerHTML='';
  cats.forEach(cat=>{
    const hs=HABITS.filter(h=>h.cat===cat);
    const catDone=hs.filter(h=>done.includes(h.id)).length;
    const catPct=Math.round(catDone/hs.length*100);
    catEl.innerHTML+=`<div style="margin-bottom:.6rem">
      <div style="display:flex;justify-content:space-between;font-size:.78rem;color:var(--text2);margin-bottom:3px">
        <span>${cat}</span><span style="color:#fff;font-weight:600">${catPct}%</span>
      </div>
      <div style="height:6px;background:var(--bg4);border-radius:99px;overflow:hidden">
        <div style="height:100%;width:${catPct}%;background:${CAT_COLOR[cat]};border-radius:99px;transition:width .4s"></div>
      </div>
    </div>`;
  });

  const slEl=document.getElementById('streakList');
  slEl.innerHTML='';
  HABITS.forEach(h=>{
    const s=getStreak(h.id);
    slEl.innerHTML+=`<div style="display:flex;align-items:center;gap:.6rem;padding:.45rem 0;border-bottom:1px solid var(--border)">
      <span style="font-size:1rem">${h.icon}</span>
      <span style="flex:1;font-size:.82rem;color:var(--text1)">${h.name}</span>
      <span style="font-size:.82rem;font-weight:700;color:var(--gold)">🔥 ${s} días</span>
    </div>`;
  });

  renderBadges(done);renderCalendar();renderWeekChart();renderSummary(done,pct);
  document.getElementById('dateLabel').textContent=new Date().toLocaleDateString('es-ES',{weekday:'long',day:'numeric',month:'long'});
}

function renderBadges(done){
  const BADGES=[
    {icon:'⭐',name:'Primer día',desc:'Completa 1 hábito',unlocked:state.xp>=10},
    {icon:'🔥',name:'En racha',desc:'3 días seguidos',unlocked:getGlobalStreak()>=3},
    {icon:'💎',name:'Semana perfecta',desc:'7 días seguidos',unlocked:getGlobalStreak()>=7},
    {icon:'🏆',name:'Perfeccionista',desc:'100% en un día',unlocked:done.length===HABITS.length},
    {icon:'📚',name:'Lector',desc:'Ambos libros al día',unlocked:done.includes('libro1')&&done.includes('libro2')},
    {icon:'💪',name:'Guerrero',desc:'Ejercicio 5 días',unlocked:Array.from({length:30},(_,i)=>new Date(Date.now()-i*864e5).toISOString().split('T')[0]).filter(k=>(state.history[k]||[]).includes('ejercicio')).length>=5},
    {icon:'🧠',name:'Mente sana',desc:'XP > 200',unlocked:state.xp>=200},
    {icon:'🌟',name:'Imparable',desc:'XP > 500',unlocked:state.xp>=500},
  ];
  document.getElementById('badgesGrid').innerHTML=BADGES.map(b=>`
    <div class="badge-item ${b.unlocked?'unlocked':'locked'}">
      <div class="badge-icon">${b.icon}</div>
      <div class="badge-name">${b.name}</div>
      <div class="badge-desc">${b.desc}</div>
    </div>`).join('');
}

function renderCalendar(){
  const now=new Date(),year=now.getFullYear(),month=now.getMonth();
  const firstDay=new Date(year,month,1).getDay(),days=new Date(year,month+1,0).getDate();
  const DAYS=['D','L','M','X','J','V','S'];
  const el=document.getElementById('calGrid');
  const monthNames=['Enero','Febrero','Marzo','Abril','Mayo','Junio','Julio','Agosto','Septiembre','Octubre','Noviembre','Diciembre'];
  document.getElementById('calMonthLabel').textContent=monthNames[month]+' '+year;
  el.innerHTML=DAYS.map(d=>`<div class="cal-day-label">${d}</div>`).join('');
  for(let i=0;i<(firstDay||7)-1;i++)el.innerHTML+=`<div></div>`;
  for(let d=1;d<=days;d++){
    const k=`${year}-${String(month+1).padStart(2,'0')}-${String(d).padStart(2,'0')}`;
    const h=state.history[k]||[];
    const isToday=k===todayKey;
    let cls='empty';
    if(h.length===HABITS.length)cls='full';
    else if(h.length>0)cls='partial';
    el.innerHTML+=`<div class="cal-day ${cls} ${isToday?'today':''}" title="${h.length}/${HABITS.length} hábitos">${d}</div>`;
  }
}

let weekChartInstance=null;
function renderWeekChart(){
  const labels=[],data=[];
  const DAY_NAMES=['Dom','Lun','Mar','Mié','Jue','Vie','Sáb'];
  for(let i=6;i>=0;i--){
    const dt=new Date(Date.now()-i*864e5);
    const k=dt.toISOString().split('T')[0];
    labels.push(DAY_NAMES[dt.getDay()]);
    data.push(Math.round((state.history[k]||[]).length/HABITS.length*100));
  }
  const ctx=document.getElementById('weekChart');
  if(weekChartInstance)weekChartInstance.destroy();
  weekChartInstance=new Chart(ctx,{
    type:'bar',
    data:{labels,datasets:[{label:'%',data,backgroundColor:data.map(v=>v===100?'#10b98190':v>=50?'#7c6af790':'#ef444460'),borderRadius:6,borderSkipped:false}]},
    options:{responsive:true,maintainAspectRatio:false,plugins:{legend:{display:false},tooltip:{callbacks:{label:c=>c.parsed.y+'%'}}},scales:{y:{min:0,max:100,ticks:{color:'#9999b3',callback:v=>v+'%'},grid:{color:'#2a2a3e'}},x:{ticks:{color:'#9999b3'},grid:{display:false}}}}

function renderSummary(done,pct){
  const el=document.getElementById('summaryText');
  if(!done.length){el.textContent='Completa hábitos hoy para ver tu resumen.';return;}
  const pending=HABITS.filter(h=>!done.includes(h.id)).map(h=>h.name);
  el.textContent=`Hoy llevas ${done.length} de ${HABITS.length} hábitos (${pct}%). ${pending.length?'Pendiente: '+pending.slice(0,2).join(', ')+(pending.length>2?'...':'')+'.' :'¡Día perfecto!'}`;
}

async function getMotivation(){
  const done=todayDone(),pct=Math.round(done.length/HABITS.length*100),streak=getGlobalStreak();
  const pending=HABITS.filter(h=>!done.includes(h.id)).map(h=>h.name);
  const el=document.getElementById('motivText'),meta=document.getElementById('motivMeta');
  el.textContent='Generando...';el.className='motivation-text loading';meta.textContent='';
  const prompt=`Eres un coach de hábitos motivador y cercano. El usuario lleva hoy ${done.length} de 9 hábitos completados (${pct}%). Su racha global es de ${streak} días. ${pending.length?'Pendientes: '+pending.join(', '):' Ha completado todos los hábitos.'}. Escríbele un mensaje motivacional corto (2-3 frases), positivo y directo. Sin asteriscos ni markdown. En español de España.`;
  try{
    const res=await fetch('https://api.anthropic.com/v1/messages',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({model:'claude-sonnet-4-20250514',max_tokens:1000,messages:[{role:'user',content:prompt}]})});
    const data=await res.json();
    el.textContent=data.content?.map(c=>c.text||'').join('')||'Sigue adelante, cada hábito cuenta.';
    el.className='motivation-text';
    meta.textContent='— Coach IA · '+new Date().toLocaleTimeString('es-ES',{hour:'2-digit',minute:'2-digit'});
  }catch(e){el.textContent='No se pudo conectar con la IA. ¡Pero tú puedes solo!';el.className='motivation-text';}
}
